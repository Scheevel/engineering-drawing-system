const getEnvBoolean = (key: string, defaultValue: boolean): boolean => {
  const value = process.env[key];
  if (value === undefined) return defaultValue;
  return value.toLowerCase() === 'true';
};
 
const getEnvNumber = (key: string, defaultValue: number): number => {
  const value = process.env[key];
  if (value === undefined) return defaultValue;
  const parsed = parseInt(value, 10);
  return isNaN(parsed) ? defaultValue : parsed;
};
 
const getEnvString = (key: string, defaultValue: string): string => {
  return process.env[key] || defaultValue;
};

  };
}

// ========================================
// CONFIGURATION MANAGER
// ========================================
 
class SchemaConfigManager {
  private config: SchemaFeatureConfig;
  private listeners: Array<(config: SchemaFeatureConfig) => void> = [];
 
  constructor(initialConfig: SchemaFeatureConfig = defaultConfig) {
    this.config = { ...initialConfig };
  }
 
  getConfig(): SchemaFeatureConfig {
    return { ...this.config };
  }
 
  updateConfig(updates: Partial<SchemaFeatureConfig>): void {
    this.config = {
      ...this.config,
      ...updates,
      autoSave: { ...this.config.autoSave, ...updates.autoSave },
      undoRedo: { ...this.config.undoRedo, ...updates.undoRedo },
      performance: { ...this.config.performance, ...updates.performance },
      validation: { ...this.config.validation, ...updates.validation },
      ui: { ...this.config.ui, ...updates.ui },
      development: { ...this.config.development, ...updates.development },
    };
 
    // Notify listeners
    this.listeners.forEach(listener => listener(this.config));
 
    // Persist to localStorage for user preferences
    this.persistUserPreferences();
  }
 
  updateAutoSaveConfig(updates: Partial<SchemaFeatureConfig['autoSave']>): void {
    this.updateConfig({ autoSave: updates });
  }
 
  updatePerformanceConfig(updates: Partial<SchemaFeatureConfig['performance']>): void {
    this.updateConfig({ performance: updates });
  }
 
  addListener(listener: (config: SchemaFeatureConfig) => void): () => void {
    this.listeners.push(listener);
 
    // Return unsubscribe function
    return () => {
      const index = this.listeners.indexOf(listener);
      if (index > -1) {
        this.listeners.splice(index, 1);
      }
    };
  }
 
  private persistUserPreferences(): void {
    try {
      const userPreferences = {
        autoSave: {
          enabled: this.config.autoSave.enabled,
          interval: this.config.autoSave.interval,
        },
        ui: {
          enableAdvancedMode: this.config.ui.enableAdvancedMode,
          defaultTheme: this.config.ui.defaultTheme,
          compactMode: this.config.ui.compactMode,
        },
        performance: {
          enableVirtualScrolling: this.config.performance.enableVirtualScrolling,
          virtualScrollThreshold: this.config.performance.virtualScrollThreshold,
        },
      };
 
      localStorage.setItem('schema_user_preferences', JSON.stringify(userPreferences));
    } catch (error) {
      console.warn('Failed to persist user preferences:', error);
    }
  }
 
  loadUserPreferences(): void {
    try {
      const stored = localStorage.getItem('schema_user_preferences');
      if (stored) {
        const userPreferences = JSON.parse(stored);
        this.updateConfig(userPreferences);
      }
    } catch (error) {
      console.warn('Failed to load user preferences:', error);
    }
  }
 
  resetToDefaults(): void {
    this.config = { ...defaultConfig };
    this.listeners.forEach(listener => listener(this.config));
    localStorage.removeItem('schema_user_preferences');
  }
 
  validateConfig(): string[] {
    const errors: string[] = [];
 
    // Validate auto-save configuration
    if (this.config.autoSave.interval < 5000) {
      errors.push('Auto-save interval must be at least 5 seconds');
    }
 
    if (this.config.autoSave.maxRetries < 0 || this.config.autoSave.maxRetries > 10) {
      errors.push('Auto-save max retries must be between 0 and 10');
    }
 
    // Validate performance configuration
    if (this.config.performance.virtualScrollThreshold < 10) {
      errors.push('Virtual scroll threshold must be at least 10 fields');
    }
 
    if (this.config.performance.debounceDelayMs < 100 || this.config.performance.debounceDelayMs > 2000) {
      errors.push('Debounce delay must be between 100ms and 2000ms');
    }
 
    // Validate undo/redo configuration
    if (this.config.undoRedo.maxStackSize < 5 || this.config.undoRedo.maxStackSize > 500) {
      errors.push('Undo/redo stack size must be between 5 and 500');
    }
 
    return errors;
  }
 
  // Feature flag helpers
  isFeatureEnabled(feature: keyof SchemaFeatureConfig, subFeature?: string): boolean {
    const featureConfig = this.config[feature];
    if (typeof featureConfig === 'boolean') {
      return featureConfig;
    }
 
    if (subFeature && typeof featureConfig === 'object') {
      return (featureConfig as any)[subFeature] ?? false;
    }
 
    return false;
  }
 
  getFeatureConfig<K extends keyof SchemaFeatureConfig>(feature: K): SchemaFeatureConfig[K] {
    return this.config[feature];
  }
}

// REACT HOOK FOR CONFIGURATION
// ========================================
 
import { useEffect, useState } from 'react';
 
export const useSchemaConfig = () => {
  const [config, setConfig] = useState<SchemaFeatureConfig>(schemaConfigManager.getConfig());
 
  useEffect(() => {
    const unsubscribe = schemaConfigManager.addListener(setConfig);
    return unsubscribe;
  }, []);
 
  return {
    config,
    updateConfig: (updates: Partial<SchemaFeatureConfig>) => schemaConfigManager.updateConfig(updates),
    updateAutoSaveConfig: (updates: Partial<SchemaFeatureConfig['autoSave']>) =>
      schemaConfigManager.updateAutoSaveConfig(updates),
    updatePerformanceConfig: (updates: Partial<SchemaFeatureConfig['performance']>) =>
      schemaConfigManager.updatePerformanceConfig(updates),
    resetToDefaults: () => schemaConfigManager.resetToDefaults(),
    validateConfig: () => schemaConfigManager.validateConfig(),
    isFeatureEnabled: (feature: keyof SchemaFeatureConfig, subFeature?: string) =>
      schemaConfigManager.isFeatureEnabled(feature, subFeature),
    getFeatureConfig: <K extends keyof SchemaFeatureConfig>(feature: K) =>
      schemaConfigManager.getFeatureConfig(feature),
  };
};
